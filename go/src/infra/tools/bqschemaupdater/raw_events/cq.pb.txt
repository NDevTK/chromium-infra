table_id: "cq"
name: "CQ Events"
description: "Events from commit queue. This was closely modeled after the JSON data that is sent to chromium-cq-status.appspot.com. All tags and certain fields, however, were excluded: owner, message, state, timestamp. Fields that have been renamed have noted this in their description."
fields: <
  name: "timestamp_millis"
  description: "Time of event, in milliseconds since the UNIX epoch."
  is_required: true
  type: INTEGER
>
fields: <
  name: "action"
  description: "Type of CQ event, e.g. PATCH_COMMITTED."
  type: STRING
>
fields: <
  name: "attempt_start_usec"
  description: "Timestamp when the CQ attempt was started (user clicked on CQ bit)."
  type: INTEGER
>
fields: <
  name: "cq_name"
  description: "Name of the CQ that sends this event."
  type: STRING
>
fields: <
  name: "issue"
  description: "Issue ID that is processed by CQ; at most 100 characters."
  type: STRING
>
fields: <
  name: "patchset"
  description: "Patchset ID that is processed by CQ; at most 100 characters."
  type: STRING
>
fields: <
  name: "dry_run"
  description: "True if CQ is in dry-run mode."
  type: BOOLEAN
>
fields: <
  name: "failure_reason"
  description: "Detailed reason on failure to commit."
  type: RECORD
  schema: <
    name: "verifier"
    type: STRING
  >
  schema: <
    name: "fail_type"
    type: STRING
  >
  schema: <
    name: "failed_try_jobs"
    is_repeated: true
    type: RECORD
    schema: <
      name: "master"
      type: STRING
    >
    schema: <
      name: "builder"
      type: STRING
    >
    schema: <
      name: "build_id"
      type: INTEGER
    >
    schema: <
      name: "fail_type"
      type: STRING
    >
  >
>

  name: "contributing_buildbucket_ids"
  description: "List of build buildbucket IDs that contributed to an attempt."
  is_repeated: true
  type: INTEGER
